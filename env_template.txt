# Azure Authentication Configuration
# Copy these values to a .env file in the same directory

# Microsoft App Registration (for user authentication)
AZURE_CLIENT_ID=your_azure_client_id_here
AZURE_CLIENT_SECRET=your_azure_client_secret_here

# Tenant IDs
# USER_TENANT_ID options:
#   - "common" = Allow any Microsoft account (personal + work/school)
#   - "organizations" = Allow any work/school account from any tenant (MULTI-TENANT - REQUIRED for Excel file users)
#   - Specific tenant ID = Only allow users from that specific tenant (SINGLE-TENANT - NOT recommended)
# 
# IMPORTANT: Since your Excel file contains users from different tenants, you MUST use "organizations" 
# to allow all users in the Excel file to login. Using a specific tenant ID will only allow users from that tenant.
USER_TENANT_ID=organizations
AZURE_AI_TENANT_ID=your_ai_tenant_id_here

# Azure AI Project
AZURE_AI_ENDPOINT=your_azure_ai_endpoint_here

# Redirect URI - must match Azure App Registration
# IMPORTANT: This must match the port your app runs on (default: 8000)
REDIRECT_URI=http://localhost:8000

# Logic App URL for submitting employee onboarding data
LOGIC_APP_SUBMIT_URL=https://prod-xx.logic.azure.com:443/workflows/YOUR_WORKFLOW_ID/triggers/manual/paths/invoke?api-version=2016-10-01&sp=%2Ftriggers%2Fmanual%2Frun&sv=1.0&sig=YOUR_SIGNATURE_HERE
